create table study_group (
                             id INT not null,
                             name text
);

create table student (
                         id INT not null,
                         surname text,
                         name text,
                         second_name text,
                         study_group_id int
);

create table subject (
                         id INT not null,
                         name text,
                         short_name text
);

create table study_plan (
                            id INT not null,
                            subject_id int,
                            exam_type_id int
);

create table journal (
                         id INT not null,
                         student_id int,
                         study_plan_id int,
                         in_time bit,
                         count int,
                         mark_id int
);

create table mark (
                      id INT not null,
                      name text,
                      value text
);

create table exam_type (
                           id INT not null,
                           type text
);
